class TrafficLight
{
private String color;
private int duration;
TrafficLight(String c,int d)
{
color=c;
duration=d;
}
String changeColor(String c,int d)
{
color=c;
duration=d;
return color;
}
boolean isRed()
{
return "red".equalsIgnoreCase(color);
}
boolean isGreen()
{
return "green".equalsIgnoreCase(color);
}
String getColor()
{
return color;
}
}
class Main
{
public static void main(String st[])
{
TrafficLight obj=new TrafficLight("red",90);
System.out.println("Current color is: "+obj.getColor());
System.out.println("After the change the color: "+
obj.changeColor("green", 30));

System.out.println("Now Color is red: "+obj.isRed());
System.out.println("Now Color is Green: "+obj.isGreen());
}
}
