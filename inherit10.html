class Employee
{
String name;
String address;
double salary;
String job;
Employee(String n,String a,double s,String j)
{
name=n;
address=a;
salary=s;
job=j;
}
double Bonus()
{
return 0;
}
String Performance()
{
return "No performance report available";
}
String Project()
{
return "No projects";
}

}
class Manager extends Employee
{
Manager(String n,String a,double s,String j)
{
super(n,a,s,j);
}
double Bonus()
{
return salary*0.15;
}
String Performance()
{
return name+" : Excellent";
}
String Project()
{
return name+" is managing a project";
}

}
