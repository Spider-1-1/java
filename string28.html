import java.util.Scanner;
class string
{
public static void main(String st[])
{
Scanner sc=new Scanner(System.in);
System.out.println("Enter the string: ");
String str=sc.nextLine();
int ch[]=new int[127];
for(int i=0;i< str.length();i++)
{
char chr=str.charAt(i);
ch[chr]++;
}
for(int i=0;i< 127;i++)
{
for(int j=i+1;j< 127;j++)
{
if(ch[i]< ch[j])
{
int temp=ch[i];
ch[i]=ch[j];
ch[j]=temp;
}
}
}
if(ch[0]==1)
{
System.out.println("No repeating sequence in a string");
}
else
{
System.out.println("Longest repeating sequence in a string is: "+ch[0]);
}
}
}
