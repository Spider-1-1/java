import java.util.Scanner;
class palin
{
public static void main(String st[])
{
Scanner sc=new Scanner(System.in);
System.out.print("Enter the number: ");
int num=sc.nextInt();
int n=num;
int sum=0;
int s=num;
while(num>0)
{
n=num%10;
sum=(sum*10)+n;
num=num/10;
}
if(s==sum)
{
System.out.print(s+" is palindrome");
}
else
{
System.out.print(s+" is not a palindrome");       
}
}
}
