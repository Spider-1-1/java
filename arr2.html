import java.util.Scanner;

class Array {
public static void main(String st[]) {
int a;
Scanner sc = new Scanner(System.in);
System.out.println("Enter the array size: ");
a = sc.nextInt();
int ar[] = new int[a];
for (int i = 0; i < ar.length; i++) {
System.out.println("Enter the array at [" + i + "] location: ");
ar[i] = sc.nextInt();
}
int visited[] = new int[ar.length];
for (int i = 0; i < ar.length; i++) {
int count = 1;
if (visited[i] == 1)
continue;
for (int j = i + 1; j < ar.length; j++) {
if (ar[i] == ar[j]) {
count++;
visited[j] = 1;
}
}
System.out.println("The Frequency of " + ar[i] + " is " + count);
}
}
}
