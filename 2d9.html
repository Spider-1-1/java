import java.util.Scanner;
class matrix
{
public static void main(String st[])
{
Scanner sc=new Scanner(System.in);
System.out.print("Enter the matrix rows: ");
int row=sc.nextInt();
System.out.print("Enter the matrix columns: ");
int col=sc.nextInt();
int ar[][]=new int[row][col];
for(int i=0;i<row;i++)
{
for(int j=0;j<col;j++)
{
System.out.print("Enter the matrix at location ["+i+"]"+"["+j+"]: ");
ar[i][j]=sc.nextInt();
}
}
int sum=0;
for(int i=0;i<row;i++)
{
for(int j=0;j<col;j++)
{
sum=sum+ar[i][j];
}
System.out.println("The sum of row "+(i+1)+" is: "+sum);
sum=0;
}
for(int j=0;j<row;j++)
{
for(int i=0;i<col;i++)
{
sum=sum+ar[i][j];
}
System.out.println("The sum of column "+(j+1)+" is: "+sum);
sum=0;
}
}
}
