interface Shape
{
void getArea();
}
class Rectangle implements Shape
{
int height;
int width;
int total;
Rectangle(int h,int w)
{
height=h;
width=w;
}
public void getArea()
{
total=height*width;
System.out.println("Area of Rectangle is: "+total);
}
}
class Circle implements Shape
{
int radious;
double total;
Circle(int r)
{
radious=r;
}
public void getArea()
{
total=3.14*radious*radious;
System.out.println("Area of Circle is: "+total);
}
}
class Triangle implements Shape
{
float base;
float height;
Triangle(float b,float h)
{
base=b;
height=h;
}
public void getArea()
{
double total=0.5*base*height;
System.out.println("Area of Triangle is :"+total);
}
}
class interfaces
{
public static void main(String st[])
{
Rectangle r=new Rectangle(4,6);
r.getArea();
System.out.println();

Circle c=new Circle(5);
c.getArea();
System.out.println();

Triangle t=new Triangle(5,6);
t.getArea();

}
}
