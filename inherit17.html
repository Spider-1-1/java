abstract class BankAccount
{
abstract void deposit(int r);
abstract void withdraw(int r);
}
class SavingsAccount extends BankAccount
{
String acc;
int amount;
SavingsAccount(String account,int amt)
{
acc=account;
amount=amt;
}
void deposit(int amt)
{
amount=amount+amt;
System.out.println("Total amount of account number "+acc+" in savings account is: "+amount+" rps.");
}
void withdraw(int amt)
{
if(amount>amt)
{
amount-=amt;
System.out.println("After withdraw the money of account number "+acc+" in savings account is: "+amount+" rps.");
}
else
{
System.out.println("Insufficient Balance");
}
}
}
