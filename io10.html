import java.io.*;
class io
{
public static void main(String st[])
{
try
{
FileReader f= new FileReader("C:\\Users\\Mohd Jahid\\OneDrive\\Desktop\\new.txt");
int i;
int line=1;

int ch=0;
while((i=f.read())!=-1)
{
System.out.print((char)i);
if((char)i=='\n')
{
line++;
}
else if(((char)i==' ' || (char)i=='\r' || (char)i=='\f' ) && (char)i!='\n')
{
word++;
}
else if(((char)i>='a' && (char)i<='z') || ((char)i>='A' && (char)i<='Z'))
{
ch++;
}
}
System.out.println("\n");
System.out.println("Total number of lines are: "+line);
System.out.println("Total number of words are: "+word);
System.out.println("Total number of characters are: "+ch); 
}
catch(Exception e)
{
System.out.println("File does not exists");
}
}
}
