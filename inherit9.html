class Truck extends Vehicle
{
double cargo;
Truck(String mak,String mod,int y,String fuel_y,double fuel_e,double capacity)
{
super(mak,mod,y,fuel_y,fuel_e);
cargo=capacity;
}
String Model()
{
return model;
}
double Fuel_efficiancy()
{
return fuel_eff*(1.0/(1.0+(cargo/1000.0)));
}
double Distance_travelled()
{
return Fuel_efficiancy()*fuel_eff;
}
double Max_speed()
{
return 80.0;
}
}
