class array
{
public static void main(String st[])
{
Scanner sc=new Scanner(System.in);
System.out.print("Enter the array size: ");
int a=sc.nextInt();
int ar[]=new int[a];
for(int i=0;i< ar.length;i++)
{
System.out.print("Enter the array at ["+i+"] location: ");
ar[i]=sc.nextInt();
}
int temp;
for(int i=0;i< ar.length;i++)
{
for(int j=i+1;j< ar.length;j++)
{
if(ar[i]>ar[j])
{
temp=ar[i];
ar[i]=ar[j];
ar[j]=temp;
}
}
}
int size=a;
for(int i=0;i< size;i++)
{
for(int j=i+1;j< size;j++)
{
if(ar[i]==ar[j])
{
for(int k=j;k< size-1;k++)
{
ar[k]=ar[k+1];
}
j--;
size--;
}

}
}
if(size>=2)
{
System.out.println("Second largest array is: "+ar[size-2]);
}
else
{
System.out.println("No Second array");
}
}
}
