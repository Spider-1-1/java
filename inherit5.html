class BankAccount
{
String account_number;
double balance;
BankAccount(String acc,double bal)
{
account_number=acc;
balance=bal;
}
void deposit(double amount)
{
if(amount>0)
{
balance=balance+amount;
}
System.out.println("Total balance is: "+balance);
}
void withdraw(double amount)
{
if(balance>=amount)
{
balance=balance-amount;
System.out.println("After withdraw the money: "+balance);
}
else
{
System.out.println("Insufficiant Balance");
}
}
}
class SavingsAccount extends BankAccount
{
SavingsAccount(String acc,double bal)
{
super(acc,bal);
}
void withdraw(double amount)
{
if(balance-amount>=100)
{
balance=balance-amount;
System.out.println("After withdraw the amount: "+balance);
}
else
{
System.out.println("Minimum balance 100 rps required");
}
}
}
class inherit
{
public static void main(String st[])
{
SavingsAccount s=new SavingsAccount("3344Bob",5335);
s.deposit(5000);
s.withdraw(6000);
s.withdraw(5000);
}
}
