import java.util.Scanner;
class matrix
{
public static void main(String st[])
{
Scanner sc=new Scanner(System.in);
System.out.print("Enter the matrix rows: ");
int row=sc.nextInt();
System.out.print("Enter the matrix columns: ");
int col=sc.nextInt();
int ar[][]=new int[row][col];
for(int i=0;i<row;i++)
{
for(int j=0;j<col;j++)
{
System.out.print("Enter the matrix at location ["+i+"]"+"["+j+"]: ");
ar[i][j]=sc.nextInt();
}
}
System.out.println("Upper Triangular Matrix is: ");
for(int i=0;i<row;i++)
{
for(int j=0;j<col;j++)
{
if(i>j)
{
System.out.print("0\t");
}
else
{
System.out.print(ar[i][j]+"\t");
}
}
System.out.println();
}
}

}
