import java.util.Scanner;
class matrix
{
public static void main(String st[])
{
Scanner sc=new Scanner(System.in);
System.out.print("Enter the matrix rows: ");
int row=sc.nextInt();
System.out.print("Enter the matrix columns: ");
int col=sc.nextInt();
int ar[][]=new int[row][col];
for(int i=0;i<row;i++)
{
for(int j=0;j<col;j++)
{
System.out.print("Enter the matrix at location ["+i+"]"+"["+j+"]: ");
ar[i][j]=sc.nextInt();
}
}
System.out.println("Original Matrix is: ");
for(int i=0;i<row;i++)
{
for(int j=0;j<col;j++)
{
System.out.print(ar[i][j]+"\t");
}
System.out.println();
}
int ar2[][]=new int[row][col];
for(int i=0;i<row;i++)
{
for(int j=0;j<col;j++)
{
ar2[i][j]=ar[j][i];
}
}
System.out.println("Transpose Matrix is: ");
for(int i=0;i<row;i++)
{
for(int j=0;j<col;j++)
{
System.out.print(ar2[i][j]+"\t");
}
System.out.println();
}
}
}
