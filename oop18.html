import java.util.ArrayList;

class Restaurant
{
ArrayList< String> menu_item=new ArrayList<>();
private int price;
ArrayList< Double> ratings=new ArrayList<>();
Restaurant(String item,int p,double rating)
{
menu_item.add(item);
price=p;
ratings.add(rating);
}
void addItem(String item,double rating)
{
if(menu_item.equals(item))
{
System.out.println("Item is already exists");
}
else
{
menu_item.add(item);
ratings.add(rating);
System.out.println("Item "+item+" is added successfully");
}
}
void removeItem(String item,double rating)
{
if(menu_item.isEmpty())
{
System.out.println("No item exists");
}
else
{
menu_item.remove(item);
ratings.remove(rating);
System.out.println("Item "+item+" is removed");
}
}
double averageRating()
{
double total=0.0;
for(Double rating: ratings)
{
total=total+rating;
}
return total/ratings.size();
}
}

class Main
{
public static void main(String st[])
{
Restaurant obj=new Restaurant("Chilli Potato",340,6.5);
obj.addItem("Frence Fries", 8.5);
obj.addItem("Burger", 9.6);

obj.removeItem("Frence Fries", 8.5);

System.out.println("Average Rating is: "+obj.averageRating());
}
}
