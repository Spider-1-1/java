boolean pass(String str)
{
if(str.length()< 8)
{
return false;
}
boolean lower=false;
boolean upper=false;
boolean digit=false;
boolean special=false;
for(int i=0;i< str.length();i++)
{
char ch=str.charAt(i);
if(islower(ch))
{
lower=true;
}
else if(isupper(ch))
{
upper=true;
}
else if(isdigit(ch))
{
digit=true;
}
else if(isspecial(ch))
{
special=true;
}
}
return upper && digit && special && lower;
}
boolean isdigit(char ch)
{
return ch>='0' && ch<='9';
}
boolean islower(char ch)
{
return ch>='a' && ch<='z';
}
boolean isupper(char ch)
{
return ch>='A' && ch<='Z';
}
boolean isspecial(char ch)
{
return ch=='!' || ch=='@' ||ch=='#' ||ch=='$' 
||ch=='%' ||ch=='^' ||ch=='&' ||ch=='*' ||ch=='(' ||ch==')' 
||ch=='~' ||ch=='/' ||ch=='<' ||ch=='>' ||ch=='}' ||ch==']';
}
