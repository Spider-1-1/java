abstract class Bird
{
abstract void fly();
abstract void makeSound();
}
class Eagle extends Bird
{
void fly()
{
System.out.println("Eagle flying high in the sky");
}
void makeSound()
{
System.out.println("Eagle Sound is: Screech Screech");
}
}
class Hawk extends Bird
{
void fly()
{
System.out.println("Hawk flying throung the air");
}
void makeSound()
{
System.out.println("Hawk Sound is: Caw Caw");
}
}
class inherit
{
public static void main(String st[])
{
Bird b=new Eagle();
b.fly();
b.makeSound();
System.out.println();

Bird b1=new Hawk();
b1.fly();
b1.makeSound();
}
}
