class Rectangle
{
private float width;
private float height;
private float a;
private float p;
Rectangle(float w,float h)
{
width=w;
height=h;
}
float Area()
{
a=width*height;
return a;
}
float Perimeter()
{
p=2*(height+width);
return p;
}
}

class Main
{
public static void main(String st[])
{
Rectangle obj=new Rectangle(3.5f,4.5f);
System.out.println("Area of rectangle is: "+obj.Area());
System.out.println("Perimeter of Rectangle is: "+obj.Perimeter());
}
}
