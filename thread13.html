class A extends Thread
{
public void run()
{
try{
for(int i=0;i< 5;i++)
{
System.out.println(getName()+"   "+i);
}
yield();
}
catch(Exception e)
{
System.out.println(e.getMessage());
}
}
}

class Main
{
public static void main(String st[]) throws InterruptedException
{
A obj=new A();
A obj2=new A();
A obj3=new A();

obj.setName("Thread 1");
obj2.setName("Thread 2");
obj3.setName("Thread 3");

obj.start();
obj2.start();
obj3.start();

obj.join();
}
}
