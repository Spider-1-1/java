import java.util.ArrayList;

class Movie
{
private String title;
private String director;
private String actor;
ArrayList< String> reviews=new ArrayList<>();
Movie(String t,String d,String a,String r)
{
title=t;
director=d;
actor=a;
reviews.add(r);
}
void addReview(String s)
{
reviews.add(s);
System.out.println("Review "+s+" is added");
}
void retreive()
{
for(String s: reviews)
{
System.out.println(s);
}
}
}

class Main
{
public static void main(String st[])
{
Movie obj=new Movie("Title","Director","Actor","Wonderful Movie");
obj.addReview("Great Movie");
obj.addReview("The best Movie in the world");

obj.retreive();
}
}
