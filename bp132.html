import java.util.Scanner;
class factorial
{
public static void main(String st[])
{
Scanner sc=new Scanner(System.in);
System.out.print("Enter the number: ");
int num=sc.nextInt();
System.out.print("Factorial is: "+fact(num));
}
static int fact(int n)
{
if(n>=1)
{
return n*fact(n-1);
}
else
{
return 1;
}
}
}
