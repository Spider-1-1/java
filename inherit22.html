INPUT

abstract class Person
{
abstract void eat();
abstract void exercise();
}
class Athlete extends Person
{
void eat()
{
System.out.println("Athlete eat food which is full of fibre,iron,calcium,etc..");
}
void exercise()
{
System.out.println("Athlete do excersice daily to build strength and endurance");
}
}
class LazyPerson extends Person
{
void eat()
{
System.out.println("LazyPerson eat food which is full of calories and high sugar");
}
void exercise()
{
System.out.println("LazyPerson is not exercise Result fatty body");
}
}
class inherit
{
public static void main(String st[])
{
Person p=new Athlete();
p.eat();
p.exercise();
System.out.println();

Person p1=new LazyPerson();
p1.eat();
p1.exercise();
}
}
