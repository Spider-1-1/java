class array
{

{

for(int i=0;i< ar.length;i++)
{
System.out.print("Enter the array at ["+i+"] location: ");

}
System.out.print("Enter the number you want to search: ");
int item=sc.nextInt();
int num=0;
for(int i=0;i< ar.length;i++)
{
if(ar[i]==item)
{
num=i+1;
}
}
if(num!=0)
{
System.out.println("Number is present at location "+num);
}
else
{
System.out.println("Number is not present");
}
}
}
