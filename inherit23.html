abstract class Instrument
{
abstract void play();
abstract void tune();
}
class Glockenspiel extends Instrument
{
void play()
{
System.out.println("Glockenspiel: Playing the notes on the metal bars");
}
void tune()
{
System.out.println("Glockspiel: Tuning the metal bars to the correct pitch");
}
}
class Violin extends Instrument
{
void play()
{
System.out.println("Violin: Playing the strings with a fingers");
}
void tune()
{
System.out.println("Violin: Tuning the Strings to the correct pitch");
}
}
class inherit
{
public static void main(String st[])
{
Instrument i=new Glockenspiel();
i.play();
i.tune();
System.out.println();

Instrument i1=new Violin();
i1.play();
i1.tune();
}
}
