class array{
public static void main(String st[])
{
int a;
Scanner sc=new Scanner(System.in);
System.out.println("Enter the array size: ");
a=sc.nextInt();
int ar[]=new int[a];
for(int i=0;i< ar.length;i++)
{
System.out.println("Enter the array at ["+i+"] location: ");
ar[i]=sc.nextInt();
}
for(int i=0;i< ar.length;i++)
{
for(int j=i+1;j< ar.length;j++)
{
if(ar[i]>ar[j])
{
int temp=ar[j];
ar[j]=ar[i];
ar[i]=temp;
}
}
}
int size=ar.length;
for(int i=0;i< size;i++)
{
for(int j=i+1;j< size;j++)
{
if(ar[i]==ar[j])
{
for(int k=j;k< size-1;k++)
{
ar[k]=ar[k+1];
}
size--;
j--;
}
}
}
System.out.println("After removing of Duplicate arrays are: ");
for(int i=0;i< size;i++)
{
System.out.println(ar[i]);
}
if(size< 2 || size< 0)
{
System.out.println("No second position in the array");
}
else
{
System.out.println("The 2nd Smallest value is: "+ar[1]);
}
}
}
