import java.util.ArrayList;

class Book
{
private String title;
private String author;
private String isbn;
ArrayList< String[]> book=new ArrayList<>();
void addBooks(String t,String a,String i)
{
title=t;
author=a;
isbn=i;
String[] b={title,author,isbn};
book.add(b);
}
void removeBooks(String isbn)
{
boolean tr=book.removeIf(b->b[2].equals(isbn));
if(tr)
{
System.out.println("Book is removed");
}
else
{
System.out.println("Can't find the book");
}

}
void show()
{
if(book.isEmpty())
{
System.out.println("There is no Books");
}
else
{
for(String[] b:book)
{
System.out.println("Title: "+b[0]+" Author: "+b[1]+" ISBN: "+b[2]);
}
}
}
}

class Main
{
public static void main(String st[])
{
Book obj=new Book();
obj.addBooks("Java", "Indu Arora","23456" );
obj.addBooks("PHP","Kalyani","34567");
obj.removeBooks("23456");
obj.show();
}
}
