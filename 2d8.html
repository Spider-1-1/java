class matrix
{

{

System.out.print("Enter the matrix columns: ");
int col=sc.nextInt();
int ar[][]=new int[row][col];
for(int i=0;i<row;i++)
{
for(int j=0;j<col;j++)
{
System.out.print("Enter the matrix at location ["+i+"]"+"["+j+"]: ");
ar[i][j]=sc.nextInt();
}
}
int count=0;
int count2=0;
for(int i=0;i<row;i++)
{
for(int j=0;j<col;j++)
{
if(ar[i][j]%2==0)
{
count++;
}
else
{
count2++;
}
}
}
System.out.println("Frequency of Even numbers are: "+count);
System.out.println("Frequency of Odd numbers are: "+count2);
}
}
