import java.util.Scanner;

class prime {
public static void main(String st[]) {
Scanner sc = new Scanner(System.in);
System.out.print("Enter the number: ");
int a = sc.nextInt();
int j = 0; 

if (a <= 1) {
j = 1; 
} else {
for (int i = 2; i < a; i++) {
if (a % i == 0) {
j = 1;
break;
}
}
}

if (j == 0) {
System.out.println("Number is prime");
} else {
System.out.println("Number is not prime");
}
}
}
