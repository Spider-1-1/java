class A extends Thread
{
int sum=0;
int n;
A(int n)
{
this.n=n;
}

public void run()
{
for(int i=0;i< n;i++)
{
if(isPrime(i))
{
sum+=i;
}
}
}

boolean isPrime(int n)
{
if(n< 2)
{
return false;
}
for(int i=2;i< n;i++)
{
if(n%i==0)
{
return false;
}
}
return true;
}

void show()
{

}
}
