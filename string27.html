class string

String str2[]=str.split("\\W+");
String temp;
int len=str2.length;
for(int i=0;i< str2.length;i++)
{
for(int j=i+1;j< str2.length;j++)
{
if(str2[i].length()>str2[j].length())
{
temp=str2[i];
str2[i]=str2[j];
str2[j]=temp;
}
}
}

System.out.println("Maximum word in a string is: "+str2[len-1]);
}
}
