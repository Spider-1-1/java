abstract class Employee
{
String name;
int salary;
String address;
Employee(String n,int s,String add)
{
name=n;
salary=s;
address=add;
}
abstract int calculate_salary();
abstract void display_info();
}
class Manager extends Employee
{
int bonus;
int total;
Manager(String n,int s,String add,int b)
{
super(n,s,add);
bonus=b;
}
int calculate_salary()
{
return bonus+salary;
}
void display_info()
{
System.out.println("Manager Name: "+name);
System.out.println("Manager Address: "+address);
System.out.println("Manager Base Salary: "+salary+" rps");
System.out.println("Manager Total Salary: "+calculate_salary()+" rps.");
}
}
class Programmer extends Employee
{
int bonus;
int total;
Programmer(String n,int s,String add,int b)
{
super(n,s,add);
bonus=b;
}
int calculate_salary()
{
return bonus+salary;
}
void display_info()
{
System.out.println("Programmer Name: "+name );
System.out.println("Programmer Address: "+address);
System.out.println("Programmer Base Salary: "+salary+ " rps");
System.out.println("Programmer Total Salary: "+calculate_salary()+" rps");
}
}
class inherit
{
public static void main(String st[])
{
Employee e=new Manager("Jahid",40000,"123 ABC",3000);
e.display_info();
System.out.println();
Employee e1=new Programmer("Harshit",30000,"234 BCD",4000);
e1.display_info();
}
}
