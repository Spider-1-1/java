import java.awt.*;
import java.awt.event.*;

class A extends Frame implements ItemListener
{
Checkbox c1,c2;
Label label;
A()
{
CheckboxGroup grp=new CheckboxGroup();

c1=new Checkbox("10th",grp,true);
c2=new Checkbox("12th",grp,false);

label=new Label();
label.setBounds(200,200,200,40);

setLayout(null);

c1.setBounds(30,30,100,30);
c2.setBounds(30,60,100,30);

add(c1);
add(c2);
add(label);

c1.addItemListener(this);
c2.addItemListener(this);

setSize(500,500);
setVisible(true);
}

public void itemStateChanged(ItemEvent e)
{
if(e.getSource()==c1)
{
label.setText("Selected box is: "+c1.getLabel());
}
else
{
label.setText("Selected box is: "+c2.getLabel());
}
}
}

class B
{
public static void main(String st[])
{
new A();
}
}
