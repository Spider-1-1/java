abstract class Shape3D
{
abstract void calculateVolume();
abstract void calculateSurfaceArea();
}
class Sphere extends Shape3D
{
int r;
double total;
Sphere(int radious)
{
r=radious;
}
void calculateVolume()
{
total=(4.0/3.0)*3.14*r*r*r;
System.out.println("Volume of a Sphere is: "+total);
}
void calculateSurfaceArea()
{
total=4*3.14*r*r;
System.out.println("Surface Area of a Sphere is: "+total);
}
}
class Cube extends Shape3D
{
int s;
double total;
Cube(int side)
{
s=side;
}
void calculateVolume()
{
total=s*s*s;
System.out.println("Volume of a Cube is:"+total);
}
void calculateSurfaceArea()
{
total=6*s*s;
System.out.println("SurfaceArea of a Cube is: "+total);
}
}
class inherit
{
public static void main(String st[])
{
Shape3D s=new Sphere(5);
s.calculateSurfaceArea();
s.calculateVolume();
System.out.println();

Shape3D s1=new Sphere(4);
s1.calculateSurfaceArea();
s1.calculateVolume();
}
}
