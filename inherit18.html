INPUT

abstract class Animal
{
abstract void eat();
abstract void sleep();
}
class Lion extends Animal
{
void eat()
{
System.out.println("Lion eat meat");
}
void sleep()
{
System.out.println("Lion sleep 18 hours a day");
}
}
class Tiger extends Animal
{
void eat()
{
System.out.println("Tiger eat meat");
}
void sleep()
{
System.out.println("Tiger sleep 16 hours a day");
}
}
class Deer extends Animal
{
void eat()
{
System.out.println("Deer eat grass");
}
void sleep()
{
System.out.println("Deer sleep 5 hours a day");
}
}
class inherit
{
public static void main(String st[])
{
Animal a=new Lion();
a.eat();
a.sleep();
System.out.println();

Animal a1=new Tiger();
a1.eat();
a1.sleep();
System.out.println();

Animal a2=new Deer();
a2.eat();
a2.sleep();
}
}
