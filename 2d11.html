{0,0,0},
{0,0,1}};
int len=ar.length;
int k=0;
for(int i=0;i<len;i++)
{
for(int j=0;j<len;j++)
{
if(i==j && ar[i][j]!=1)
{
k=1;
break;
}
else if(i!=j && ar[i][j]!=0)
{
k=1;
break;
}
}
}
if(k==0)
{
System.out.println("Its an identity matrix");
}
else
{
System.out.println("Its not an identity matrix");
}
}
}
