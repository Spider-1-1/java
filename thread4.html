class A extends Thread
{
int[][] a1;
int[][] a2;
int[][] a3;
A(int[][] a,int[][] b)
{
a1=a.clone();
a2=b.clone();
a3=new int[a.length][b.length];
}
public void run()
{
for(int i=0;i< a1.length;i++)
{
for(int j=0;j< a1.length;j++)
{
a3[i][j]=0;
for(int k=0;k< a1.length;k++)
{
a3[i][j]+=a1[i][k]*a2[k][j];
}
}
}
}

void show()
{
for(int i=0;i< a3.length;i++)
{
for(int j=0;j< a3.length;j++)
{
System.out.print(a3[i][j]+"\t");
}
System.out.println();
}
System.out.println();
System.out.println();
}
}

class Main
{
public static void main(String st[])
{
int[][] a={
{1,2,3},
{4,5,6},
{7,8,9}
};
int[][] b={
{9,8,7},
{6,5,4},
{3,2,1}
};
A obj=new A(a,b);
A obj2=new A(a,b);
A obj3=new A(a,b);

obj.start();
obj2.start();
obj3.start();

obj.show();
obj2.show();
obj3.show();
}
}
