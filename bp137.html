class convert {

public static void main(String st[]) {
String Roman = "LXVII";
int result = romanToInt(Roman);
System.out.println("The Integer value of given Roman number is: " + result);
}

public static int romanToInt(String str) {
int sum = 0;
int length = str.length();

for (int i = 0; i < length; i++) {
char currentChar = str.charAt(i);
int currentValue = getValue(currentChar);

if (i + 1 < length) {
char nextChar = str.charAt(i + 1);
int nextValue = getValue(nextChar);

if (currentValue >= nextValue) {
sum += currentValue;
} else {
sum -= currentValue;
}
} else {
sum += currentValue;
}
}

return sum;
}

public static int getValue(char roman) {
switch (roman) {
case 'I': return 1;
case 'V': return 5;
case 'X': return 10;
case 'L': return 50;
case 'C': return 100;
case 'D': return 500;
case 'M': return 1000;
default: return 0;
}
}
}
