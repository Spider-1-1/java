import java.applet.Applet;
import java.awt.*;
import java.util.Calendar;
import java.util.Date;

public class applet14 extends Applet implements Runnable
{
Thread t;
Font font;
Color color;
public void init()
{
setBackground(Color.YELLOW);
t=null;
font=new Font("Algerian",Font.ROMAN_BASELINE,50);
color = Color.RED;
}
public void start()
{
if(t==null)
{
t=new Thread(this);
t.start();
}
}
public void stop()
{
t=null;
}
public void run()
{
while(Thread.currentThread()==t)
{
repaint();
try
{
Thread.currentThread().sleep(1000);
}
catch(InterruptedException e)
{
System.out.println(e);
}

}
}
public void paint(Graphics g)
{
Calendar calendar=Calendar.getInstance();
Date day=calendar.getTime();
g.setFont(font);
g.setColor(color);
g.drawString(day+"\t ", 20,60 );
showStatus("This is the Watch.");
}
}
