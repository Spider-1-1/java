class Employee
{
private String name;
private String job;
private int salary;
void setName(String n)
{
name=n;
}
void setJob(String j)
{
job=j;
}
void setSalary(int s)
{
salary=s;
}
String getName()
{
return name;
}
String getJob()
{
return job;
}
int getSalary()
{
return salary;
}
int calculate(int sal)
{
salary=salary+sal;
return salary;
}
}

class Main
{
public static void main(String st[])
{
Employee obj=new Employee();
obj.setName("abc");
obj.setJob("Typist");
obj.setSalary(1000);
System.out.println("Name is: "+obj.getName());
System.out.println("Job Title: "+obj.getJob());
System.out.println("Current Salary: "+obj.getSalary());
System.out.println("After Updating The Salary: "+obj.calculate(5000));
}
}
