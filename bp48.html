import java.util.Scanner;
class batting
{
public static void main(String st[])
{
Scanner sc=new Scanner(System.in);
System.out.print("Enter the total runs: ");
int run=sc.nextInt();
System.out.print("Enter the total matches: ");
int match=sc.nextInt();
System.out.print("Enter the total not-out matches: ");
int not=sc.nextInt();
int dismissed=match-not;
double bat=run/dismissed;
System.out.println("The Batting average is: "+bat);
}

}
